            int |$\proghole{}{.}$| (int |$a$|, int |$b$|){
                assume((0 < |$a$|) && (|$a$| < |$b$|));
                while (|$a$| < |$b$|) {
                    |$c$| = |$c$| + |\hole{\cdot}|;
                    |$a$| = |$a$| + 1;
                }
                return |$c$|;
            }
            // |POST: \colorbox{green!20!white}{$(c_2 > c_1)$}|
        \end{minted}
